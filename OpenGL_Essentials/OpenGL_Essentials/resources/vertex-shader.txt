#version 400 core 
layout (location = 0) in vec3 vert_position; 
layout (location = 1) in vec3 color;
out vec3 vert_color;
uniform mat4 projection;
uniform mat4 camera;
uniform mat4 model;

void main() {
   gl_Position = projection * camera * model * vec4(vert_position, 1.0);
   vert_color = color;
}
